#version 330 core
layout (location = 0) in vec3 aPos;
uniform vec2 world_scale;
uniform vec2 offset;
void main() {
   vec3 world = aPos + vec3(offset, 0);
   vec3 screen_offbyhalf = world * vec3(world_scale, 1);
   vec3 screen = screen_offbyhalf + vec3(-1.0, -1.0, 0);
   gl_Position = vec4(screen, 1.0);
}
